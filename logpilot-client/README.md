## ğŸ›  Usage

```ts
import { LogPilotClient } from "logpilot-lite-client";

const client = new LogPilotClient(process.env.LOGPILOT_SERVER_URL || "localhost:50051");

client.send({
  channel: 'job',
  level: 'ERROR',
  message: 'Token cleanup failed',
  meta: { error: 'Refresh token not found' },
  storage: 'sqlite'
});
```
